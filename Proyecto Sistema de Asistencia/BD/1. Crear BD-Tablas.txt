CREATE DATABASE BDWeb_Sistema_Asistencia
GO

USE BDWeb_Sistema_Asistencia
GO

CREATE TABLE TB_ROLES (
Id_Rol INT IDENTITY (1,1) NOT NULL PRIMARY KEY,
Descripcion VARCHAR(50) NULL,
Fecha_Creacion_Registro DATETIME NULL DEFAULT GETDATE(),
Estado BIT NULL DEFAULT 1
)
GO

CREATE TABLE TB_USUARIO (
Id_Usuario INT IDENTITY (1,1) NOT NULL PRIMARY KEY,
Nombres   VARCHAR(50) NULL,
Apellidos VARCHAR(50) NULL,
Celular   VARCHAR(12) NULL,
Correo VARCHAR(100) NULL UNIQUE,
Contrasenia VARBINARY(255) NULL,
Id_Rol INT NOT NULL FOREIGN KEY REFERENCES dbo.TB_ROLES,
Imagen VARCHAR(MAX) NULL,
Fecha_Creacion_Registro DATETIME NULL DEFAULT GETDATE(),
Estado BIT NULL DEFAULT 1
)
GO


CREATE TABLE TB_ASISTENCIA (
Id_Asistencia INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
Id_Usuario INT NOT NULL FOREIGN KEY REFERENCES dbo.TB_USUARIO,
Fecha_Inicio DATETIME NULL,
Fecha_Fin DATETIME NULL,
flag_Asistencia BIT	NULL DEFAULT 0
)
GO
